    <!-- Firebase SDKs -->

        // Firebase Config

        const firebaseConfig = typeof __firebase_config !== 'undefined' 

            ? JSON.parse(__firebase_config) 

            : { /* Add your Firebase config object here if not in Canvas */ };

            

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';


        // Initialize Firebase

        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);

        

        // DOM Elements and State

        const filterBtns = document.querySelectorAll('.filter-btn');

        const loadingSpinner = document.getElementById('loading-spinner');

        

        let allProducts = []; // Cache to store all products

        let isAdmin = false; // Track admin login state

        let currentFilter = 'all'; // Track current filter


        /**

         * Shows a custom alert message.

         */

        function showAlert(message, isError = false) {

            const customAlert = document.getElementById('custom-alert');

            const alertMessage = document.getElementById('alert-message');

            alertMessage.textContent = message;

            customAlert.className = 'fixed top-5 right-5 text-white py-3 px-5 rounded-lg shadow-lg z-50 animate-pulse';

            customAlert.classList.add(isError ? 'bg-red-500' : 'bg-green-500');

            setTimeout(() => customAlert.classList.add('hidden'), 3000);

        }


        /**

         * Renders products to the grid, with admin controls if logged in.

         */

        function renderProducts(products) {

            if (!productGrid) return;

            productGrid.innerHTML = ''; 

            if (products.length === 0) {

                 productGrid.innerHTML = `<p class="text-center text-gray-500 col-span-full">No products found in this category.</p>`;

                 return;

            }

            products.forEach(product => {

                const card = document.createElement('div');

                card.className = 'product-card bg-white rounded-lg overflow-hidden shadow-lg border border-pink-100 relative';

                card.innerHTML = `

                    ${isAdmin ? `<button data-id="${product.id}" class="delete-btn absolute top-2 right-2 bg-red-600 text-white rounded-full w-8 h-8 flex items-center justify-center hover:bg-red-700 z-10"><i class="fas fa-trash-alt"></i></button>` : ''}

                    <img src="${product.imageUrl}" alt="${product.name}" class="w-full h-64 object-cover" onerror="this.onerror=null;this.src='https://placehold.co/600x800/f871b1/ffffff?text=Image+Not+Found';">

                    <div class="p-5">

                        <h3 class="text-xl font-semibold mb-2 truncate">${product.name}</h3>

                        <p class="text-pink-600 font-bold text-2xl mb-4">₹${product.price}</p>

                        <button class="w-full bg-pink-500 text-white font-bold py-2 px-4 rounded-lg hover:bg-pink-600 transition-all duration-300 flex items-center justify-center">

                            <i class="fab fa-whatsapp mr-2"></i>Enquire on WhatsApp

                        </button>

                    </div>

                `;

                productGrid.appendChild(card);

            });

        }

        

        /**

         * Applies the current filter to the products list and re-renders it.

         */

        function applyFilter() {

             if (currentFilter === 'all') {

                renderProducts(allProducts);

            } else {

                const filteredProducts = allProducts.filter(p => p.category === currentFilter);

                renderProducts(filteredProducts);

            }

        }

        

        /**

         * Fetches and displays products from Firestore in real-time.

         */

        function fetchAndDisplayProducts() {

            loadingSpinner.style.display = 'block';

            const productsCollectionPath = `/artifacts/${appId}/public/data/products`;

            const productsCollection = collection(db, productsCollectionPath);

            

            onSnapshot(query(productsCollection), (snapshot) => {

                allProducts = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

                allProducts.sort((a, b) => (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0));

                applyFilter(); // Apply current filter after fetching

                loadingSpinner.style.display = 'none';

            }, error => {

                console.error("Error fetching products: ", error);

                showAlert("Could not fetch products. Please try again later.", true);

                loadingSpinner.style.display = 'none';

            });

        }

        /**

         * Signs the user in, either with a custom token or anonymously.

         */

        async function signInUser() {

            try {

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                    console.log("Signed in with custom token.");

                } else {

                    await signInAnonymously(auth);

                    console.log("Signed in anonymously.");

                }

            } catch (error) {

                console.error("Authentication failed:", error);

                showAlert("Authentication failed. Cannot load data.", true);

            }

        }


        // Event Listeners

        adminLoginBtn.addEventListener('click', () => {

            const password = prompt('Please enter the admin password:');

            if (password === 'satnam123') {

                isAdmin = true;

                adminModal.classList.add('active');

                applyFilter(); // Re-render products with admin controls

                showAlert('Admin login successful!', false);

            } else if (password !== null) {

                showAlert('Incorrect password!', true);

            }

        });


        closeModalBtn.addEventListener('click', () => adminModal.classList.remove('active'));

        window.addEventListener('click', (e) => {

            if (e.target === adminModal) adminModal.classList.remove('active');

        });


        addProductForm.addEventListener('submit', async (e) => {

            e.preventDefault();

            const newProduct = {

                name: document.getElementById('product-name').value,

                price: Number(document.getElementById('product-price').value),

                category: document.getElementById('product-category').value,

                createdAt: serverTimestamp()

            };


            const productsCollectionPath = `/artifacts/${appId}/public/data/products`;

            await addDoc(collection(db, productsCollectionPath), newProduct);

            showAlert('Product added successfully!', false);

            addProductForm.reset();

        productGrid.addEventListener('click', async (e) => {

           if(e.target.closest('.delete-btn')) {

               const btn = e.target.closest('.delete-btn');

               const id = btn.dataset.id;

               if (id && confirm('Are you sure you want to delete this product?')) {

                   try {

                       const productDocPath = `/artifacts/${appId}/public/data/products/${id}`;

                       await deleteDoc(doc(db, productDocPath));
